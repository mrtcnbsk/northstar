```ts
// kilocode_change start - SP2 Task 1: evaluator panel view-model. Loop defaults mirror the
// LOCKED schema defaults (Org.loop.maxIterations=4, evaluatorModel="haiku"); used whenever the
// SP1 detail response omits the resolved `loop` block (older run / degraded org read).
export const DEFAULT_MAX_ITERATIONS = 4
export const DEFAULT_EVALUATOR_MODEL = "haiku"

export type VerdictView = { pass: boolean; reasons?: string[]; ts: string }

/** Minimal structural shape of one per-stage view under SP1's extended detail response — satisfied
 * by the real `OrgRunStageView` without coupling this module to the SDK package (mirrors
 * `StageStatusView` above). */
export type EvaluatorStageView = {
  stage: string
  status: string
  criteria?: string[]
  iterations?: number
  verdictHistory?: VerdictView[]
}

export type EvaluatorDetailView = {
  run: { status: string; pausedReason?: { kind: string; stage: string; detail: string } | null }
  stages: EvaluatorStageView[]
  loop?: { maxIterations: number; evaluatorModel: string }
}

export type EvaluatorCriterion = { text: string; met: boolean }

export type EvaluatorPanel = {
  stage: string | null
  criteria: EvaluatorCriterion[]
  iteration: number
  maxIterations: number
  latestRejection: string | null
  /** Latest verdict's overall pass; null when the stage has no verdict yet. */
  passed: boolean | null
}

/** The stage the panel focuses: a paused run points straight at its pausedReason.stage; otherwise
 * the first running stage, else the first awaiting_approval stage (pipeline order preserved by the
 * caller-supplied `stages`). */
function focusStage(detail: EvaluatorDetailView): EvaluatorStageView | undefined {
  const paused = detail.run.pausedReason?.stage
  if (paused) {
    const s = detail.stages.find((st) => st.stage === paused)
    if (s) return s
  }
  return detail.stages.find((s) => s.status === "running") ?? detail.stages.find((s) => s.status === "awaiting_approval")
}

/** Verdict has no per-criterion data (LOCKED Verdict = {pass, reasons?, summary?}), so a criterion
 * is deemed unmet ONLY when a rejection reason name-references it (case-insensitive substring);
 * unreferenced criteria stay met, so a failing verdict never blanket-fails every criterion. A
 * passing verdict marks all met; no verdict yet marks all unmet. */
function criterionMet(text: string, verdict: VerdictView | undefined): boolean {
  if (!verdict) return false
  if (verdict.pass) return true
  const needle = text.toLowerCase()
  return !(verdict.reasons ?? []).some((r) => r.toLowerCase().includes(needle))
}

export function buildEvaluatorPanel(detail: EvaluatorDetailView): EvaluatorPanel {
  const maxIterations = detail.loop?.maxIterations ?? DEFAULT_MAX_ITERATIONS
  const stage = focusStage(detail)
  if (!stage) {
    return { stage: null, criteria: [], iteration: 0, maxIterations, latestRejection: null, passed: null }
  }
  // kilocode_change - SP2: `.at(-1)` yields `VerdictView | undefined` cleanly (no [len-1] index
  // access), and the reason is read via `(latest?.reasons ?? [])[0] ?? fallback` so `latestRejection`
  // is a plain `string | null`, matching the declared EvaluatorPanel type under any index-access mode.
  const latest = (stage.verdictHistory ?? []).at(-1)
  const criteria = (stage.criteria ?? []).map((text) => ({ text, met: criterionMet(text, latest) }))
  const firstReason = (latest?.reasons ?? [])[0]
  const latestRejection = latest && !latest.pass ? firstReason ?? "rejected (no reason given)" : null
  return {
    stage: stage.stage,
    criteria,
    iteration: stage.iterations ?? 0,
    maxIterations,
    latestRejection,
    passed: latest ? latest.pass : null,
  }
}
// kilocode_change end
```

```ts
// kilocode_change start - SP2 Task 2: loop gauge view-model.
export type LoopStageView = { stage: string; status: string; iterations?: number; startedAt?: string | null }
export type LoopDetailView = {
  run: { createdAt: string; status: string; pausedReason?: { kind: string; stage: string; detail: string } | null }
  stages: LoopStageView[]
  loop?: { maxIterations: number; evaluatorModel: string }
}

export type LoopGaugeVM = {
  iteration: number
  maxIterations: number
  elapsed: string
  evaluatorModel: string
  atLimit: boolean
}

/** "45s" / "2m 05s" / "1h 02m 05s". Negative / non-finite input clamps to "0s" so a clock skew or
 * an unparseable timestamp can never render NaN. */
export function formatElapsed(ms: number): string {
  const safe = Number.isFinite(ms) && ms > 0 ? ms : 0
  const total = Math.floor(safe / 1000)
  const h = Math.floor(total / 3600)
  const m = Math.floor((total % 3600) / 60)
  const s = total % 60
  const pad = (n: number) => String(n).padStart(2, "0")
  if (h > 0) return `${h}h ${pad(m)}m ${pad(s)}s`
  if (m > 0) return `${m}m ${pad(s)}s`
  return `${s}s`
}

export function loopGauge(detail: LoopDetailView, now: number = Date.now()): LoopGaugeVM {
  const maxIterations = detail.loop?.maxIterations ?? DEFAULT_MAX_ITERATIONS
  const evaluatorModel = detail.loop?.evaluatorModel ?? DEFAULT_EVALUATOR_MODEL
  const active = detail.stages.find((s) => s.status === "running") ?? detail.stages.find((s) => s.status === "awaiting_approval")
  const iteration = active?.iterations ?? 0
  const startedStr = active?.startedAt ?? detail.run.createdAt
  const started = startedStr ? Date.parse(startedStr) : Number.NaN
  const elapsed = Number.isNaN(started) ? formatElapsed(0) : formatElapsed(now - started)
  return { iteration, maxIterations, elapsed, evaluatorModel, atLimit: iteration >= maxIterations }
}
// kilocode_change end
```

```ts
// kilocode_change - new file
/**
 * SP2 Task 4: pure conversation-strip builders. `conversationCard` picks AT MOST one card from the
 * SP1 detail response; `parseMention` extracts a leading `@name` for org_note targeting. Kept
 * SDK-free (structural input) so both are unit-tested with in-memory fixtures — the strip component
 * (`mission-strip.tsx`) renders the result and `CockpitView` dispatches from it via `sdk.client.orgRuns.*`.
 */
export type ConversationVerdict = { pass: boolean; reasons?: string[]; ts: string }

export type ConversationStageView = {
  stage: string
  status: string
  criteria?: string[]
  verdictHistory?: ConversationVerdict[]
}

export type ConversationDetailView = {
  run: { status: string; auto?: boolean; pausedReason?: { kind: string; stage: string; detail: string } | null }
  stages: ConversationStageView[]
}

export type ConversationCard =
  | { kind: "none" }
  | { kind: "plan"; stage: string; criteria: string[] }
  | { kind: "escalation"; stage: string; reasons: string[]; detail: string }
  | { kind: "final_gate"; stage: string; detail: string }

/**
 * Priority: a PAUSED run's pausedReason drives the escalation / final-gate card (the conductor
 * parked here awaiting a human). Otherwise, a run not yet handed to the conductor (`auto !== true`)
 * showing an awaiting_approval stage drives the plan-approval card. Everything else is composer-only
 * (`none`) — notably an AUTO run's mid-loop gate is server-resolved by the conductor, not a card.
 */
export function conversationCard(detail: ConversationDetailView): ConversationCard {
  const { run } = detail
  if (run.status === "paused" && run.pausedReason) {
    const { kind, stage, detail: reason } = run.pausedReason
    if (kind === "escalation") {
      const s = detail.stages.find((x) => x.stage === stage)
      // kilocode_change - SP2: `.at(-1)` yields `ConversationVerdict | undefined` cleanly (no
      // [len-1] index access); `latest.reasons ?? []` keeps `reasons` a plain `string[]`.
      const latest = s?.verdictHistory?.at(-1)
      const reasons = latest && !latest.pass ? latest.reasons ?? [] : []
      return { kind: "escalation", stage, reasons, detail: reason }
    }
    if (kind === "final_gate") return { kind: "final_gate", stage, detail: reason }
  }
  if (run.auto !== true) {
    const gate = detail.stages.find((s) => s.status === "awaiting_approval")
    if (gate) return { kind: "plan", stage: gate.stage, criteria: gate.criteria ?? [] }
  }
  return { kind: "none" }
}

/** `@name rest` -> {target:name, text:rest}; anything else -> broadcast {target:"*", text:trimmed}.
 * Destructures the match with a guard so `target`/`text` are plain `string` (never `string | undefined`)
 * regardless of the compiler's index-access strictness. */
export function parseMention(input: string): { target: string; text: string } {
  const trimmed = input.trim()
  const m = MENTION.exec(trimmed)
  if (m) {
    const [, name, rest] = m
    if (name && rest) return { target: name, text: rest.trim() }
  }
  return { target: "*", text: trimmed }
}
```
